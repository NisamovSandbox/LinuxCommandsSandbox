=== xargs
Construir y ejecutar comandos a partir de la entrada estándar.

[source,bash]
----
# Uso básico de xargs
echo "archivo1 archivo2" | xargs rm
----

-n:: `xargs -n` Número de argumentos por ejecución

[source,bash]
----
# Ejecutar el comando con un argumento por vez
echo "a b c" | xargs -n 1 echo
----

-I:: `xargs -I` Reemplazar argumentos con un marcador

[source,bash]
----
# Usar {} como marcador de posición
ls *.txt | xargs -I {} cp {} backup/{}
----

-0:: `xargs -0` Usar separador nulo

[source,bash]
----
# Manejar nombres con espacios
find . -name "*.log" -print0 | xargs -0 rm
----

-p:: `xargs -p` Confirmar antes de ejecutar

[source,bash]
----
# Pedir confirmación al usuario
echo "archivo1 archivo2" | xargs -p rm
----

-t:: `xargs -t` Mostrar el comando antes de ejecutarlo

[source,bash]
----
# Mostrar comando ejecutado
echo "a b c" | xargs -t echo
----

-r:: `xargs -r` No ejecutar si no hay entrada

[source,bash]
----
# Evitar ejecuciones sin argumentos
grep "error" logs.txt | xargs -r rm
----

-max-procs:: `xargs -P` Ejecución en paralelo

[source,bash]
----
# Ejecutar hasta 4 procesos en paralelo
echo {1..10} | xargs -n 1 -P 4 sleep
----
