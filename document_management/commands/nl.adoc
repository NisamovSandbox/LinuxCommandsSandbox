=== nl
Numerar líneas de un fichero o de la entrada estándar.

[source,bash]
----
# Numerar todas las líneas de un archivo
nl fichero.txt
----

[unordered.stack]
-b:: `nl -b` Especificar qué líneas numerar
+
[source,bash]
----
# Numerar solo líneas no vacías
nl -b a fichero.txt
----

[unordered.stack]
-v:: `nl -v` Número inicial
+
[source,bash]
----
# Comenzar numeración desde 100
nl -v 100 fichero.txt
----

[unordered.stack]
-i:: `nl -i` Incremento entre números de línea
+
[source,bash]
----
# Incrementar de 5 en 5
nl -i 5 fichero.txt
----

[unordered.stack]
-n:: `nl -n` Formato del número
+
[source,bash]
----
# Formato con ceros a la izquierda
nl -n rz fichero.txt
----

[unordered.stack]
-s:: `nl -s` Separador entre número y línea
+
[source,bash]
----
# Usar “->” como separador
nl -s '->' fichero.txt
----

[unordered.stack]
-w:: `nl -w` Ancho del número de línea
+
[source,bash]
----
# Ancho de 4 caracteres
nl -w 4 fichero.txt
----
