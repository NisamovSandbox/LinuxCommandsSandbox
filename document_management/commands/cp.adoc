=== cp
Copiar archivos o directorios.

[source,bash]
----
# Copiar un fichero a un directorio destino
cp fichero.txt /ruta/destino/
----
[unordered.stack]
-r:: `cp -r` Copiar recursivamente el objetivo

[source,bash]
----
# Copiar recursivamente un directorio
cp -r carpeta_origen/ carpeta_destino/
----
[unordered.stack]
-f:: `cp -f` Forzar la copia sobrescribiendo sin preguntar

[source,bash]
----
# Forzar copia
cp -f fichero_origen fichero_destino
----
[unordered.stack]
-a:: `cp -a` Copiar archivos manteniendo atributos (recursivo)

[source,bash]
----
# Copiar preservando atributos
cp -a carpeta_origen/ carpeta_destino/
----
[unordered.stack]
-b:: `cp -b` Crear copia de seguridad del destino

[source,bash]
----
# Copiar con copia de seguridad
cp -b fichero.txt fichero.txt.bak
----
[unordered.stack]
-i:: `cp -i` Pedir confirmación antes de sobrescribir

[source,bash]
----
# Copiar solicitando confirmación
cp -i fichero.txt /ruta/destino/
----
[unordered.stack]
-v:: `cp -v` Mostrar archivos mientras se copian (verbose)

[source,bash]
----
# Copiar mostrando archivos procesados
cp -v fichero.txt /ruta/destino/
----
[unordered.stack]
--preserve:: `cp --preserve=ATTRS` Preservar atributos (mode,ownership,timestamps)

[source,bash]
----
# Copiar preservando permisos y timestamps
cp --preserve=mode,timestamps fichero.txt /ruta/destino/
----
[unordered.stack]
--no-preserve:: `cp --no-preserve=ATTRS` No preservar ciertos atributos

[source,bash]
----
# Copiar sin preservar propietario
cp --no-preserve=ownership fichero.txt /ruta/destino/
----
[unordered.stack]
--parents:: `cp --parents` Mantener la ruta de origen al copiar

[source,bash]
----
# Copiar creando la ruta relativa en destino
cp --parents dir/a/b/file.txt /backup/
----
[unordered.stack]
--reflink:: `cp --reflink[=always|never]` Intentar copia por referencia (CoW)

[source,bash]
----
# Intentar copia por referencia cuando sea posible
cp --reflink=auto fichero.bin /ruta/destino/
----
