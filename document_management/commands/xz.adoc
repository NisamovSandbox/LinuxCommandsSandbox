=== xz
Comprimir y descomprimir archivos usando el algoritmo LZMA/LZMA2.

[source,bash]
----
# Comprimir un archivo
xz fichero.txt
----

[unordered.stack]
-d:: `xz -d` Descomprimir archivos .xz

[source,bash]
----
# Descomprimir un archivo
xz -d fichero.txt.xz
----

[unordered.stack]
-k:: `xz -k` Mantener el archivo original

[source,bash]
----
# Comprimir sin borrar el archivo original
xz -k fichero.txt
----

[unordered.stack]
-l:: `xz -l` Mostrar información del archivo comprimido

[source,bash]
----
# Ver detalles del archivo .xz
xz -l fichero.txt.xz
----

[unordered.stack]
-c:: `xz -c` Enviar salida a stdout

[source,bash]
----
# Comprimir y mostrar por salida estándar
xz -c fichero.txt > fichero.txt.xz
----

[unordered.stack]
-t:: `xz -t` Comprobar integridad del archivo

[source,bash]
----
# Verificar que el archivo no está corrupto
xz -t fichero.txt.xz
----

[unordered.stack]
-0-9:: `xz -0` a `xz -9` Nivel de compresión

[source,bash]
----
# Máxima compresión
xz -9 fichero.txt
----

[unordered.stack]
-z:: `xz -z` Forzar compresión (opción por defecto)

[source,bash]
----
# Forzar compresión explícitamente
xz -z fichero.txt
----
