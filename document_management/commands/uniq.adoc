=== uniq
Eliminar lineas duplicadas consecutivas.

[source,bash]
----
# Eliminar líneas duplicadas consecutivas
uniq entrada.txt > salida.txt
----

[unordered.stack]
-c:: `uniq -c` Mostrar numero de repeticiones en columna
----
[source,bash]
----
# Contar repeticiones de líneas consecutivas
uniq -c entrada.txt
----

[unordered.stack]
-d:: `uniq -d` Mostrar lineas duplicadas
----
[source,bash]
----
# Mostrar sólo las líneas duplicadas consecutivas
uniq -d entrada.txt
----

[unordered.stack]
-u:: `uniq -u` Mostrar lineas no duplicadas
----
[source,bash]
----
# Mostrar sólo líneas únicas
uniq -u entrada.txt
----

[unordered.stack]
-i:: `uniq -i` Ignorar mayúsculas/minúsculas
----
[source,bash]
----
# Ignorar case
uniq -i entrada.txt
----

[unordered.stack]
-f:: `uniq -f N` Ignorar las primeras N columnas (palabras)
----
[source,bash]
----
# Ignorar la primera columna al comparar
uniq -f 1 entrada.txt
----

[unordered.stack]
-s:: `uniq -s N` Ignorar los primeros N caracteres
----
[source,bash]
----
# Ignorar primeros 3 caracteres
uniq -s 3 entrada.txt
----
