=== updatedb
Actualizar la base de datos utilizada por el comando locate.

[source,bash]
----
# Actualizar la base de datos de locate
updatedb
----

[unordered.stack]
-U:: `updatedb -U` Definir directorios a indexar

[source,bash]
----
# Indexar únicamente /home
updatedb -U /home
----

[unordered.stack]
-o:: `updatedb -o` Especificar archivo de base de datos

[source,bash]
----
# Crear base de datos personalizada
updatedb -o /tmp/mi_locate.db
----

[unordered.stack]
--prunepaths:: `updatedb --prunepaths` Excluir rutas

[source,bash]
----
# Excluir /proc y /sys
updatedb --prunepaths="/proc /sys"
----

[unordered.stack]
--prunefs:: `updatedb --prunefs` Excluir sistemas de ficheros

[source,bash]
----
# Excluir sistemas virtuales
updatedb --prunefs="proc sysfs tmpfs"
----

[unordered.stack]
--prunenames:: `updatedb --prunenames` Excluir nombres concretos

[source,bash]
----
# Excluir directorios .git
updatedb --prunenames=".git"
----

[unordered.stack]
--localpaths:: `updatedb --localpaths` Indexar solo sistemas locales

[source,bash]
----
# Evitar sistemas remotos
updatedb --localpaths
----

[unordered.stack]
--debug:: `updatedb --debug` Mostrar información de depuración

[source,bash]
----
# Ver proceso detallado
updatedb --debug
----
